  Future<void> addCountriesToPassport() async {
    final FirebaseFirestore firestore = FirebaseFirestore.instance;

    Map<String, bool> countryData = {
      for (var country in countries) country: false,
    };

    try {
      await firestore.collection('medals').doc('passport').set(countryData);
      print('Countries added successfully!');
    } catch (e) {
      print('Error adding countries: $e');
    }
  }

   ElevatedButton(
          onPressed: () async {
            await addCountriesToPassport();
          },
          style: ElevatedButton.styleFrom(
            backgroundColor: Colors.green,
          ),
          child: Text(
            'Add Countries to Passport',
            style: Theme.of(context).textTheme.displaySmall,
          ),
